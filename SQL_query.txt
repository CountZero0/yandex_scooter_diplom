Задание 1.
SELECT c.login,
       COUNT(DISTINCT(o."courierId"))
FROM "Orders" AS o
       INNER JOIN "Couriers" AS c ON o."courierId" = c.id 
WHERE "inDelivery" = True
GROUP BY c.login;
Задание 2.
SELECT DISTINCT(o.track),
       CASE
           WHEN o.finished = true THEN '2'
           WHEN o.cancelled = true THEN '-1'
           WHEN o."inDelivery" = true THEN '1'
           ELSE '0'
        END AS status
FROM "Orders" AS o;
